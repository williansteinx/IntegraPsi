
CREATE SEQUENCE integrapsi.tb_usuario_id_usuario_seq;

CREATE TABLE integrapsi.tb_usuario (
                id_usuario INTEGER NOT NULL DEFAULT nextval('integrapsi.tb_usuario_id_usuario_seq'),
                ds_email VARCHAR(255) NOT NULL,
                password VARCHAR(255) NOT NULL,
                role VARCHAR(20) NOT NULL,
                created_at TIMESTAMP NOT NULL,
                update_at TIMESTAMP NOT NULL,
                CONSTRAINT pk_usuario PRIMARY KEY (id_usuario)
);


ALTER SEQUENCE integrapsi.tb_usuario_id_usuario_seq OWNED BY integrapsi.tb_usuario.id_usuario;

CREATE TABLE integrapsi.tb_psicologo (
                id_psicologo INTEGER NOT NULL,
                nm_psicologo VARCHAR(255) NOT NULL,
                crp_profissional VARCHAR(50) NOT NULL,
                created_at TIMESTAMP NOT NULL,
                update_at TIMESTAMP NOT NULL,
                CONSTRAINT pk_psicologo PRIMARY KEY (id_psicologo)
);


CREATE SEQUENCE integrapsi.tb_ativacao_paciente_id_ativacao_seq;

CREATE TABLE integrapsi.tb_ativacao_paciente (
                id_ativacao INTEGER NOT NULL DEFAULT nextval('integrapsi.tb_ativacao_paciente_id_ativacao_seq'),
                id_psicologo INTEGER NOT NULL,
                cd_token VARCHAR(10) NOT NULL,
                ds_email VARCHAR(255) NOT NULL,
                fl_utilizado BOOLEAN DEFAULT FALSE NOT NULL,
                cd_expiracao TIMESTAMP NOT NULL,
                created_at TIMESTAMP NOT NULL,
                CONSTRAINT pk_ativacao PRIMARY KEY (id_ativacao)
);


ALTER SEQUENCE integrapsi.tb_ativacao_paciente_id_ativacao_seq OWNED BY integrapsi.tb_ativacao_paciente.id_ativacao;

CREATE TABLE integrapsi.tb_paciente (
                id_paciente INTEGER NOT NULL,
                id_psicologo INTEGER NOT NULL,
                nm_paciente VARCHAR NOT NULL,
                cpf_paciente VARCHAR(11) NOT NULL,
                telefone VARCHAR(20) NOT NULL,
                dt_nascimento DATE NOT NULL,
                created_at TIMESTAMP NOT NULL,
                update_at TIMESTAMP NOT NULL,
                CONSTRAINT pk_paciente PRIMARY KEY (id_paciente)
);


CREATE SEQUENCE integrapsi.tb_agendamento_id_consulta_seq;

CREATE TABLE integrapsi.tb_agendamento (
                id_consulta INTEGER NOT NULL DEFAULT nextval('integrapsi.tb_agendamento_id_consulta_seq'),

                id_paciente INTEGER NOT NULL,

                CONSTRAINT pk_agendamento PRIMARY KEY (id_consulta)

ALTER SEQUENCE integrapsi.tb_agendamento_id_consulta_seq OWNED BY integrapsi.tb_agendamento.id_consulta;

CREATE SEQUENCE integrapsi.tb_evolucoes_id_evolucao_seq;
